示例代码

<!--引入jsAPI的功能js-->
<script src="https://as.alipayobjects.com/g/component/antbridge/1.1.1/antbridge.min.js"></script>
<!--分享功能8.1以上钱包支持-->
<script>
if(navigator.userAgent.indexOf("AlipayClient")===-1){
    alert('请在支付宝钱包内运行');
}else{
    if((Ali.alipayVersion).slice(0,3)>=8.1){
        Ali.share({
            //渠道名称。支持以下几种：Weibo/LaiwangContacts/LaiwangTimeline/Weixin/WeixinTimeLine/SMS/CopyLink
            'channels': [{
                    name: 'ALPContact',   //支付宝联系人,9.0版本
                    param: {   //请注意，支付宝联系人仅支持一下参数
                      contentType: 'text',    //必选参数,目前支持支持"text","image","url"格式
                      content:"分享内容",    //必选参数,分享描述
                      iconUrl:"https://i.alipayobjects.com/common/favicon/favicon.ico",   //必选参数,缩略图url，发送前预览使用,
                      imageUrl:"https://i.alipayobjects.com/common/favicon/favicon.ico", //图片url
                      url:"http://alipay.com",   //必选参数，卡片跳转连接
                      title:"分享的标题",    //必选参数,分享标题
                      memo:""   //透传参数,分享成功后，在联系人界面的通知提示。
                    }
                },{
                  name: 'ALPTimeLine', //支付宝生活圈
                  param: {
                    contentType: 'url',    //必选参数,目前只支持"url"格式
                    title: '分享的标题',   //标题
                    url: 'http://alipay.com',  //url
                    iconUrl:"https://i.alipayobjects.com/common/favicon/favicon.ico" //icon
                  }
                },{
                name: 'Weibo', //新浪微博
                param: {
                    title: '分享的标题',
                    content: '分享的内容，不能超过140',
                    imageUrl: '分享的图片地址',
                    captureScreen: true, //分享当前屏幕截图(和imageUrl同时存在时，优先imageUrl)
                    url: 'http://alipay.com' //分享跳转的url，当添加此参数时，分享的图片大小不能超过32K
                }
            }, {
                name: 'Weixin', //微信
                param: {
                    title: '分享的标题',
                    content: '分享的内容',
                    imageUrl: '分享的图片地址',
                    captureScreen: true,
                    url: 'http://alipay.com'
                }
            }, {
                name: 'WeixinTimeLine', //微信朋友圈
                param: {
                    title: '分享的标题',
                    content: '分享的内容',
                    imageUrl: '分享的图片地址',
                    captureScreen: true,
                    url: 'http://alipay.com'
                }
            }, {
                name: 'SMS', //短信
                param: {
                    content: '短信内容',
                    //应业务方需求定制功能
                    contentType: 'url',
                    extData:''
                }
            }, {
                name: 'CopyLink', //复制链接
                param: {
                    url: 'http://m.alipay.com'
                }
            }]
        }, function(result) {
            if(result.errorCode){
                //没有成功分享的情况
                //errorCode=10，分享失败或取消
            }else{
                //成功分享的情况
            }
        });
    }else{
        Ali.alert({
            title: '亲',
            message: '请升级您的钱包到最新版',
            button: '确定'
        });
    }
}
</script>
