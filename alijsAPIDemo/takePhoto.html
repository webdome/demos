<!--引入jsAPI的功能js-->
<script src="https://as.alipayobjects.com/g/component/antbridge/1.1.1/antbridge.min.js"></script>
<!--拍照功能8.1以上钱包支持-->
<script>
if(navigator.userAgent.indexOf("AlipayClient")===-1){
    alert('请在支付宝钱包内运行');
}else{
    if((Ali.alipayVersion).slice(0,3)>=8.1){
        Ali.photo({
            dataType: 'dataurl',
            allowedEdit: true,
            src: 'camera',
            format: 'jpg',
            quality: 100
        }, function(result) {
            if(result.errorCode){
                //没有拍照的情况
                //errorCode=10，用户取消
                //errorCode=11，操作失败（权限不够）
            }else{
                //成功拍照的情况
                //result.photo	string	照片信息，为 dataUrl 或者 fileUrl
            }
        });
    }else{
        Ali.alert({
            title: '亲',
            message: '请升级您的钱包到最新版',
            button: '确定'
        });
    }
}
</script>
