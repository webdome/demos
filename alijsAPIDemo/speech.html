语音识别结果显示在此

示例代码

<!--引入jsAPI的功能js-->
<script src="https://as.alipayobjects.com/g/component/antbridge/1.1.1/antbridge.min.js"></script>
<!--语音转文字功能8.3以上钱包支持-->
<script>
if(navigator.userAgent.indexOf("AlipayClient")===-1){
    alert('请在支付宝钱包内运行');
}else{
    if((Ali.alipayVersion).slice(0,3)>=8.3){
        Ali.isSpeechAvailable(function(result){
            // true 可用， false 不可用
            if(result.success){
                Ali.startSpeech({
                    "iOSKey": '21812030',          //key 区分平台
                    "androidKey": '21812030'
                });
                document.addEventListener('speechRecognizer',function(event){
                    if(event.data.error){
                        //没有语音识别成功的情况
                        //10	服务不可用
                        //11	无权限访问设备录音	IOS
                        //12	设备不支持蓝牙4.0
                        //20	语音识别失败
                        //21	录音失败
                        //22	没有识别到内容
                        //23	用户取消
                        //18	启动过于频繁	IOS
                        //19	网络通信异常	IOS
                        switch(result.error){
                            default:
                                Z('#J_doing').html('操作失败');
                                alert('操作失败');
                        }
                        Z('#J_done').hide();
                        Z('#J_doing').show();
                    }else if(event.data.result){
                        //语音识别成功的情况
                        Z('#J_doing').hide().html('语音识别结果显示在此');
                        Z('#J_done').html('你刚才说了：'+event.data.result);
                        Z('#J_done').show();
                    }
                },false);
            }else{
                //语音不可用
            }
        )};
    }else{
        Ali.alert({
            title: '亲',
            message: '请升级您的钱包到最新版',
            button: '确定'
        });
    }
}
</script>
