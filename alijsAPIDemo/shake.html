    if((Ali.alipayVersion).slice(0,3)>=8.2){
        Ali.shake.watch(function() {
            //配合震动api
            Ali.vibration.vibrate({
                duration: 3000
            }, function() {
                //成功震动的情况
            });
            Z("#J_doing").hide();
            Z('#J_done').html('shake,shake （配合震动api）');
            Z('#J_done').show();
        });
    }else{
        Ali.alert({
            title: '亲',
            message: '请升级您的钱包到最新版',
            button: '确定'
        });
    }
}
</script>
